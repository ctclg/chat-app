# Set variables (replace these values with your chosen names)
RESOURCE_GROUP="chatapp-rg"
LOCATION="westus2"  # or choose a location closer to you
APP_NAME="tluschatapp"  # example: "mychatapp" (must be globally unique)

# Create resource group
az group create --name $RESOURCE_GROUP --location $LOCATION

# Create App Service plan
az appservice plan create \
    --name $APP_NAME-plan \
    --resource-group $RESOURCE_GROUP \
    --sku B1 \
    --is-linux

# Create web app
az webapp create \
    --resource-group $RESOURCE_GROUP \
    --plan $APP_NAME-plan \
    --name $APP_NAME \
    --runtime "PYTHON:3.12" \
    --startup-file "startup.sh"
